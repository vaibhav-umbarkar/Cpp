class Vehicle {
public:
    void start() {
        std::cout << "Vehicle started!" << std::endl;
    }
};

class Car : public Vehicle {
public:
    void honk() {
        std::cout << "Car horn honked!" << std::endl;
    }
};

int main() {
    Car myCar;
    myCar.start();  // Inherited from Vehicle
    myCar.honk();   // Specific to Car
    return 0;
}
